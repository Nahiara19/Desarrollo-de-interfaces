### Ejemplo de GridLaout:
```python
import sys
from PyQt6.QtWidgets import QApplication, QMainWindow, QWidget, QGridLayout,QPushButton
from PyQt6.QtGui import QPalette, QColor

class MainWindow(QMainWindow):

    def __init__(self):
        super(MainWindow, self).__init__()

        self.setWindowTitle("My App")

        b1 = QPushButton("B1")
        b2 = QPushButton("B2")
        b3 = QPushButton("B3")
        b4 = QPushButton("B4")

        layout = QGridLayout()

        layout.addWidget(b1, 0, 0, 1, 1)  # fila, columna, rowSpan, columnSpan
        layout.addWidget(b2, 0, 1, 1, 1)
        layout.addWidget(b3, 0, 2, 1, 1)
        layout.addWidget(b4, 1, 0, 1, 3)  # en este caso, el widget se expandir치 en 3 columnas

        widget = QWidget()
        widget.setLayout(layout)
        self.setCentralWidget(widget)

app = QApplication(sys.argv)
window = MainWindow()
window.show()

app.exec()
```


### Ejemplo LayoutVertical:

```python
import sys
from PyQt6.QtWidgets import QApplication, QMainWindow, QWidget, QVBoxLayout,QPushButton
from PyQt6.QtGui import QPalette, QColor

class MainWindow(QMainWindow):

    def __init__(self):
        super(MainWindow, self).__init__()

        self.setWindowTitle("My App")

        layout = QVBoxLayout() #instancia del qvbox

        boton = QPushButton("Aceptar")
        layout.addWidget(boton)

        boton2 =  QPushButton("Cancelar")
        layout.addWidget(boton2)# a침adimos al layout el boton que queremos que se muestre

         # addwidget es para a침adir elementos 

        widget = QWidget() # instanciamos el qwidget para poder hacer que el layout y los botones se vean
        widget.setLayout(layout)# a침adimos  el layout al qwidget 
        self.setCentralWidget(widget)



app = QApplication(sys.argv)
window = MainWindow()
window.show()

app.exec()
```
